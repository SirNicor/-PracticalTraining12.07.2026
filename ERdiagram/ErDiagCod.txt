
@startchen

entity Groups {
    ID
    NameGroup : string
}

entity Teachers {
    ID
    FIO : string
}

entity Classrooms {
    ID
    NumberClassRooms : string
}

entity Subjects {
    ID
    NameSubjects : string
}

entity Lesson {
    ID
    DateStart : TimeOnly
    DayOfWeek : int      
    NumberOfWeer : int
    Lesson -1- Teachers
    Lesson -1- Subjects
    Lesson -1- Classrooms
}

relationship LessonGroup {
    Lesson -N- LessonGroup
    LessonGroup -N- Groups
}
relationship GroupSubject {
    Groups -N- GroupSubject
    GroupSubject -N- Subjects
}

relationship TeacherSubject {
    Teachers -N- TeacherSubject
    TeacherSubject -N- Subjects
}
@endchen
